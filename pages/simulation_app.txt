import streamlit as st
import pandas as pd
import numpy as np

st.set_page_config(page_title="Mugenuni Grow-Out Simulation", layout="wide")

# High-Contrast Styling
st.markdown("""
    <style>
    html, body, [class*="ViewContainer"] { font-size: 20px !important; }
    h1 { font-size: 44px !important; color: #000000 !important; border-bottom: 3px solid #000; }
    .stMetric { background-color: #ffffff; padding: 25px; border: 4px solid #000000; border-radius: 15px; }
    </style>
    """, unsafe_allow_html=True)

# --- SIDEBAR: SIMULATION CONTROLS ---
st.sidebar.header("🧬 BIOLOGICAL DRIVERS")
# Updated to default to the 17-week empirical average
growth_rate = st.sidebar.slider("Average Weekly Growth (mm)", 1.50, 4.00, 2.65, step=0.05)
feed_rate = st.sidebar.slider("Daily Feed Rate (% of Body Weight)", 2.0, 8.0, 5.0)

st.sidebar.divider()
st.sidebar.header("📐 RACEWAY SPECS")
start_size = 15.0 # mm
target_size = 60.0 # mm
urchins_per_raceway = 1770 # Based on 40% cover logic

# --- SIMULATION ENGINE ---
weeks = []
diameters = []
weights = []
weekly_feed = []

current_size = start_size
current_week = 0

# Run loop until they hit the 60mm enhancement threshold
while current_size < target_size:
    weeks.append(current_week)
    diameters.append(current_size)
    
    # Allometric weight estimation scaled for T. gratilla
    estimated_weight_g = 0.0004 * (current_size ** 3) 
    weights.append(estimated_weight_g)
    
    # Feed calculation for the whole raceway
    raceway_biomass_kg = (estimated_weight_g * urchins_per_raceway) / 1000
    feed_kg_per_week = raceway_biomass_kg * (feed_rate / 100) * 7
    weekly_feed.append(feed_kg_per_week)
    
    # Advance time
    current_size += growth_rate
    current_week += 1

# Create the master dataframe
sim_df = pd.DataFrame({
    "Week": weeks,
    "Diameter (mm)": diameters,
    "Weight per Urchin (g)": weights,
    "Ulva Needed (kg/wk)": weekly_feed
})

total_weeks = current_week
total_ulva = sum(weekly_feed)
final_biomass = (weights[-1] * urchins_per_raceway) / 1000

# --- MAIN INTERFACE ---
st.title("🌊 MUGENUNI: GROW-OUT SIMULATION")
st.write(f"### Tracking a single raceway from **15 mm** to **60 mm**")

col1, col2, col3 = st.columns(3)
col1.metric("TIME TO 60mm (WEEKS)", f"{total_weeks}")
col2.metric("TOTAL ULVA CONSUMED (KG)", f"{total_ulva:,.0f}")
col3.metric("FINAL STANDING BIOMASS", f"{final_biomass:,.0f} kg")

st.divider()

tab1, tab2 = st.tabs(["📈 GROWTH & FEED TRAJECTORY", "📊 WEEK-BY-WEEK DATA"])

with tab1:
    st.write("### Urchin Growth Curve (Diameter vs. Weight)")
    st.line_chart(sim_df.set_index("Week")[["Diameter (mm)", "Weight per Urchin (g)"]])
    
    st.write("### Escalating Ulva Demand (kg per week)")
    st.bar_chart(sim_df.set_index("Week")["Ulva Needed (kg/wk)"])

with tab2:
    st.write("### Raw Simulation Data")
    # Formatting for clean display
    display_df = sim_df.copy()
    display_df["Diameter (mm)"] = display_df["Diameter (mm)"].round(1)
    display_df["Weight per Urchin (g)"] = display_df["Weight per Urchin (g)"].round(1)
    display_df["Ulva Needed (kg/wk)"] = display_df["Ulva Needed (kg/wk)"].round(1)
    st.dataframe(display_df.set_index("Week"), use_container_width=True)
